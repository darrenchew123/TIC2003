procedure main {
    read a;
    read b;
    call calculateGCD;
    call printGCD;
}

procedure calculateGCD{

    gcd = 0;

    if a == 0 then {
        gcd = b;
    } else if b == 0 then {
        gcd = a;
    } else {
        while a != b {
            if a > b then {
                a = a - b;
            } else {
                b = b - a;
            }
        }
        gcd = a;
    }
}

procedure printGCD{
    print gcd;
}
